# Define your pipelines here with scalers and regressors
pipelines = {
    'Linear Regression': Pipeline([
        ('scaler', StandardScaler()),
        ('regressor', LinearRegression())
    ]),
    'Lasso Regression': Pipeline([
        ('scaler', StandardScaler()),
        ('regressor', Lasso(random_state=42))
    ]),
    'Ridge Regression': Pipeline([
        ('scaler', StandardScaler()),
        ('regressor', Ridge(random_state=42))
    ]),
    'SVR': Pipeline([
        ('scaler', StandardScaler()),
        ('regressor', SVR())
    ]),
    'Random Forest Regression': Pipeline([
        ('scaler', StandardScaler()),
        ('regressor', RandomForestRegressor(random_state=42))
    ]),
    'XGBoost Regression': Pipeline([
        ('scaler', StandardScaler()),
        ('regressor', xgb.XGBRegressor(random_state=42))
    ])
}

# Define the hyperparameter grid for each model
param_grids = {
    'Linear Regression': { # Linear regression doesn't have hyperparameters we're tuning here
    },
    'Lasso Regression': {
        'regressor__alpha': np.logspace(-4, 4, 5)
    },
    'Ridge Regression': {
        'regressor__alpha': np.logspace(-4, 4, 5)
    },
    'SVR': {
        'regressor__C': [0.1, 1, 10],
        'regressor__gamma': ['scale', 'auto']
    },
    'Random Forest Regression': {
        'regressor__n_estimators': [50, 100, 200],
        'regressor__max_depth': [5, 10, 15]
    },
    'XGBoost Regression': {
        'regressor__n_estimators': [100, 200],
        'regressor__max_depth': [3, 5, 7],
        'regressor__learning_rate': [0.01, 0.1]
    }
}

# Fit the grid search for each model
best_models = {}
for name, pipeline in pipelines.items():
    grid_search = GridSearchCV(pipeline, param_grids[name], cv=5, scoring='neg_mean_squared_error', n_jobs=-1)
    grid_search.fit(X_train, y_train)
    best_models[name] = grid_search.best_estimator_
    print(f"Best parameters for {name}: {grid_search.best_params_}")
    print(f"Best cross-validation score for {name}: {grid_search.best_score_}")

# Evaluate and output the performance of each model
for name, model in best_models.items():
    y_pred = model.predict(X_test)
    mse = mean_squared_error(y_test, y_pred)
    r2 = r2_score(y_test, y_pred)
    print(f"{name} Performance:")
    print(f"Mean Squared Error: {mse}")
    print(f"R-squared: {r2}\n")

import matplotlib.pyplot as plt
import numpy as np

# Names of models and metrics
models = ['Random Forest', 'XGBoost', 'SVR']
metrics = ['R-squared', 'MAE', 'MSE', 'RMSE']
training_scores = [
    [0.9802, 0.7054, 0.8296, 0.9108],  # Random Forest
    [0.7566, 2.5504, 10.2000, 3.1937],  # XGBoost
    [0.7860, 1.7702, 8.9685, 2.9947]    # SVR
]
test_scores = [
    [0.5301, 3.6107, 20.9765, 4.5800],  # Random Forest
    [0.5629, 3.3595, 19.5138, 4.4174],  # XGBoost
    [0.8330, 1.6335, 7.4578, 2.7309]    # SVR
]

n_metrics = len(metrics)
n_models = len(models)
index = np.arange(n_models)
bar_width = 0.35

# Create a figure and a set of subplots in 2 rows
fig, axes = plt.subplots(nrows=2, ncols=2, figsize=(10, 10), sharex='col')  # Adjust for two columns if needed

# Flatten the axes array for easy iteration
axes = axes.flatten()

# Plot each metric
for i, ax in enumerate(axes):
    if i >= n_metrics:  # In case there are more subplots than metrics, skip the extra ones
        break
    training_vals = [score[i] for score in training_scores]
    test_vals = [score[i] for score in test_scores]
    
    training_bars = ax.bar(index, training_vals, bar_width, label='Training')
    test_bars = ax.bar(index + bar_width, test_vals, bar_width, label='Test')

    # Add some labels and title
    ax.set_title(metrics[i])
    ax.set_xticks(index + bar_width / 2)
    ax.set_xticklabels(models)
    ax.legend()

# Add an overall title and labels for the axes
fig.suptitle('Model Performance Comparison')
fig.text(0.5, 0.04, 'Models', ha='center')  # x-label
fig.text(0.04, 0.5, 'Metric Values', va='center', rotation='vertical')  # y-label

# Optional: Add data labels to the bars
for ax in axes:
    for bars in ax.containers:
        ax.bar_label(bars)

plt.tight_layout(rect=[0.03, 0.03, 1, 0.95])
plt.savefig('Results2.jpg')# Adjust the padding between and around subplots
plt.show()
